// ========== 渲染逻辑 ==========
function buildFilterOptions() {
  const nations = [...new Set(PLAYERS.map(p => p.nation))].sort();
  const clubs   = [...new Set(PLAYERS.map(p => p.club))].sort();

  const nationSel = document.getElementById('filterNation');
  const teamSel   = document.getElementById('filterTeam');

  nationSel.innerHTML = '<option value="">🌍 全部国家</option>';
  nations.forEach(n => {
    const p = PLAYERS.find(x => x.nation === n);
    nationSel.innerHTML += `<option value="${n}">${p.flag} ${n}</option>`;
  });

  teamSel.innerHTML = '<option value="">🏟️ 全部俱乐部</option>';
  clubs.forEach(c => {
    teamSel.innerHTML += `<option value="${c}">${c}</option>`;
  });
}

function render() {
  const search   = document.getElementById('searchInput').value.toLowerCase().trim();
  const nation   = document.getElementById('filterNation').value;
  const position = document.getElementById('filterPos').value;
  const club     = document.getElementById('filterTeam').value;

  let filtered = PLAYERS.filter(p => {
    if (nation && p.nation !== nation) return false;
    if (position && p.pos !== position) return false;
    if (club && p.club !== club) return false;
    if (search) {
      return (
        p.cn.toLowerCase().includes(search) ||
        p.en.toLowerCase().includes(search) ||
        p.nation.includes(search) ||
        p.club.toLowerCase().includes(search) ||
        p.wcHistory.toLowerCase().includes(search)
      );
    }
    return true;
  });

  // 默认按身价降序
  filtered.sort((a, b) => b.value - a.value);

  document.getElementById('statsBar').textContent =
    `共 ${filtered.length} 位球员`;

  const grid = document.getElementById('grid');
  if (!filtered.length) {
    grid.innerHTML = '<div class="no-result">😕 没有找到匹配的球员，试试其他关键词</div>';
    return;
  }

  grid.innerHTML = filtered.map(p => {
    const initials = p.cn[0];
    const valText = p.value >= 10000
      ? `${(p.value / 10000).toFixed(1)}亿€`
      : `${p.value}万€`;

    return `
<div class="card">
  <div class="card-top">
    <div class="avatar">${initials}</div>
    <div class="player-info">
      <div class="player-name">${p.cn}</div>
      <div class="player-name-en">${p.en}</div>
      <div class="player-meta">
        <span class="tag tag-flag">${p.flag} ${p.nation}</span>
        <span class="tag tag-pos">${p.pos === 'FW' ? '前锋' : p.pos === 'MF' ? '中场' : p.pos === 'DF' ? '后卫' : '门将'}</span>
        <span class="tag tag-club">${p.club}</span>
      </div>
    </div>
  </div>

  <div class="value-row">
    <div>
      <div class="value-amount">${valText}</div>
      <div class="value-label">德转身价 · ${p.age}岁</div>
    </div>
  </div>

  <table class="detail-table">
    <tr><td>📏 身高</td><td>${p.height}cm</td></tr>
    <tr><td>⚖️ 体重</td><td>${p.weight}kg</td></tr>
    <tr><td>🏟 本/往届出场</td><td>${p.wcApps} 场</td></tr>
    <tr><td>⚽ 世界杯进球</td><td>${p.wcGoals} 球</td></tr>
    <tr><td>🎯 世界杯助攻</td><td>${p.wcAssists} 次</td></tr>
    <tr><td>🏅 主要荣誉</td><td style="font-size:0.78rem">${p.awards}</td></tr>
  </table>

  <div class="wc-history">
    <div class="wc-history-title">🏆 世界杯旅程</div>
    <div style="font-size:0.82rem; color: var(--text-secondary);">${p.wcHistory}</div>
  </div>
</div>`;
  }).join('');
}

// 初始化
buildFilterOptions();
render();

  if ((e.ctrlKey || e.metaKey) && e.key === 'k') {
    e.preventDefault();
    document.getElementById('searchInput').focus();
  }
});
